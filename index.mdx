```bash cURL wrap
# 1. Start a check, then redirect the person to the returned url.
curl -X POST https://trust.verifyyou.com/v3/initialize \
  -H "Authorization: Bearer $VY_SK" \
  -H "Content-Type: application/json" \
  -d '{"origin":"https://yourapp.com","return_path":"/verified"}'
# => { "url": "https://app.verifyyou.com/v/..." }

# 2. When they come back, confirm the token and gate on "verified".
curl https://trust.verifyyou.com/v3/confirmations/$TOKEN \
  -H "Authorization: Bearer $VY_SK"
# => { "verified": true }
```

```ts Node wrap
// 1. Start a check, then send the person to `url`.
const { url } = await fetch("https://trust.verifyyou.com/v3/initialize", {
  method: "POST",
  headers: { Authorization: `Bearer ${process.env.VY_SK}`, "Content-Type": "application/json" },
  body: JSON.stringify({ origin: "https://yourapp.com", return_path: "/verified" }),
}).then((r) => r.json());

// 2. When they return, confirm the token and gate on `verified`.
const { verified } = await fetch(`https://trust.verifyyou.com/v3/confirmations/${token}`, {
  headers: { Authorization: `Bearer ${process.env.VY_SK}` },
}).then((r) => r.json());
```

```python Python wrap
import os, requests

# 1. Start a check, then send the person to the returned url.
r = requests.post("https://trust.verifyyou.com/v3/initialize",
    headers={"Authorization": f"Bearer {os.environ['VY_SK']}"},
    json={"origin": "https://yourapp.com", "return_path": "/verified"})
url = r.json()["url"]

# 2. When they return, confirm the token and gate on "verified".
res = requests.get(f"https://trust.verifyyou.com/v3/confirmations/{token}",
    headers={"Authorization": f"Bearer {os.environ['VY_SK']}"})
verified = res.json()["verified"]
```
