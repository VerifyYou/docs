```ts Node wrap
const res = await fetch("https://trust.verifyyou.com/v3/initialize", {
  method: "POST",
  headers: {
    Authorization: `Bearer ${process.env.VY_SK}`,
    "Content-Type": "application/json",
  },
  body: JSON.stringify({ origin: "https://yourapp.com", return_path: "/verified" }),
});
const { url } = await res.json(); // redirect the user to url
```

```python Python wrap
import requests

res = requests.post(
    "https://trust.verifyyou.com/v3/initialize",
    headers={"Authorization": f"Bearer {VY_SK}"},
    json={"origin": "https://yourapp.com", "return_path": "/verified"},
)
url = res.json()["url"]  # redirect the user to url
```

```go Go wrap
body := strings.NewReader(`{"origin":"https://yourapp.com","return_path":"/verified"}`)

res, _ := http.DefaultClient.Do(req)
// decode { "url": ... } and redirect the user there
```

```ruby Ruby wrap
res = Net::HTTP.post(
```
